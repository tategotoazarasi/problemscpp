87 {

102 bool TrieNode::dfs(TrieNode* root, const string& str, int start, bool flag)
103 {
104 if (this->ch == 0)
105 {
106 //根节点
107 auto node = this->nexts[str[start] - 'a'];
108 if (node == nullptr)
109 {
110 return false;
111 }
112 if (node->dfs(root, str, start, flag))
113 {
114 return true;
115 }
116 return false;
117 }
118 //非根节点
119 //到一个单词结束处
120 if (this->is_end)
121 {
122 if (start == str.length() - 1)
123 {
124 return flag;
125 }
126 auto res = root->dfs(root, str, start + 1, true);
127 if (res)
128 {
129 return true;
130 }
131 }
132 TrieNode* node = nullptr;
133 if (str[start + 1] - 'a' >= 0)
134 {
135 node = this->nexts[str[start + 1] - 'a'];
136 }
137 return node != nullptr && node->dfs(root, str, start + 1, flag);
138 }

628 int Solution::getResult(int mouse, int cat, int turns)
629 {
630 if (turns == n * 2)
631 {
632 return DRAW;
633 }
634 if (dp[mouse][cat][turns] < 0)
635 {
636 if (mouse == 0)
637 {
638 dp[mouse][cat][turns] = MOUSE_WIN;
639 }
640 else if (cat == mouse)
641 {
642 dp[mouse][cat][turns] = CAT_WIN;
643 }
644 else
645 {
646 getNextResult(mouse, cat, turns);
647 }
648 }

652 void Solution::getNextResult(int mouse, int cat, int turns)
653 {
654 int curMove = turns % 2 == 0 ? mouse : cat;
655 int defaultResult = curMove == mouse ? CAT_WIN : MOUSE_WIN;
656 int result = defaultResult;
657 for (int next : graph[curMove])
658 {
659 if (curMove == cat && next == 0)
660 {
661 continue;
662 }
663 int nextMouse = curMove == mouse ? next : mouse;
664 int nextCat = curMove == cat ? next : cat;
665 int nextResult = getResult(nextMouse, nextCat, turns + 1);
666 if (nextResult != defaultResult)
667 {
668 result = nextResult;
669 if (result != DRAW)
670 {
671 break;
672 }
673 }
674 }
675 dp[mouse][cat][turns] = result;
