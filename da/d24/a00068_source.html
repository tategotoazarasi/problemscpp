392 TreeNode *rebuild(vector<int> &inorder, int in_begin, int in_end, vector<int> &preorder, int pre_begin, int pre_end) {
393 if(in_begin == in_end) {
394 return new TreeNode(inorder[in_begin]);
395 }
396 TreeNode *root = new TreeNode(preorder[pre_begin]);
397
398 int left_in_cursor = in_begin;
399 int left_pre_cursor = pre_begin + 1;
400
401 if(inorder[in_begin] == preorder[pre_begin]) {
402 root->left = nullptr;
403 } else {
404 unordered_set<int> left = unordered_set<int>();
405 while(inorder[left_in_cursor] != preorder[pre_begin]) {
406 left.insert(inorder[left_in_cursor]);
407 left_in_cursor++;
408 }
409 while(left.find(preorder[left_pre_cursor]) != left.end()) {
410 left_pre_cursor++;
411 }
412 root->left = rebuild(inorder, in_begin, left_in_cursor - 1, preorder, pre_begin + 1, left_pre_cursor - 1);
413 }
414
415 if(inorder[in_end] == preorder[pre_begin]) {
416 root->right = nullptr;
417 } else {
418 root->right = rebuild(inorder, left_in_cursor + 1, in_end, preorder, left_pre_cursor, pre_end);
419 }
420
421 return root;
422 }

457 // 插入操作
458 void insert(TreeNode *&root, int x) {
459 /*
460 1. 递归找到x的待插入的位置
461 2. 如果x < 当前root就递归到左子树，反之，递归到右子树。
462 */
463 if(!root)
464 root = new TreeNode(x);
465 // 如果发现数值相同的话就判断一下;
466 if(root->val == x)
467 return;
468 else if(x < root->val)
469 insert(root->left, x);
470 else
471 insert(root->right, x);
472 }
473
474 void remove(TreeNode *&root, int x) {
475 /*
476 1. 待删除的结点只有左子树。立即可推出，左子树上的结点都是小于待删除的结点的，我们只需要把待删除结点删了然后左子树接上待删除结点的父节点就可以了。
477 2. 待删除的结点只有右子树。立即可推出，右子树上的结点都是大于待删除的结点的，我们只需要把待删除结点删了然后右子树接上待删除结点的父节点就可以了。
478 3. 待删除的结点既有左子树又有右子树。这个比上两个情况麻烦一点，但也不麻烦，需要读者理解的是，怎么能删除结点后还不改变中序遍历的结果，并且操作代价最小，显而易见，我们根据待删除结点的左子树可以得到最右下角的最后结点满足$<x$并且最接近x的结点，把这个结点覆盖待删除的结点，然后把这个结点删除，就完成了我们的操作。
479 */
480
481 // 如果不存在直接return
482 if(!root)
483 return;
484 if(x < root->val)
485 remove(root->left, x);
486 else if(x > root->val)
487 remove(root->right, x);
488 else {
489 if(!root->left && !root->right)
490 root = NULL;
491 else if(!root->left)
492 root = root->right;
493 else if(!root->right)
494 root = root->left;
495 else {
496 auto p = root->left;
497 while(p->right)
498 p = p->right;
499 root->val = p->val;
500 remove(root->left, p->val);
501 }
502 }
503 }

505 // 输出数值 x 的前驱(前驱定义为现有所有数中小于 x 的最大的数)。
506 int get_pre(TreeNode *root, int x) {
507 if(!root)
508 return -INF;
509 if(root->val >= x)
510 return get_pre(root->left, x);
511 else
512 return max(root->val, get_pre(root->right, x));
513 }
514
515 // 输出数值 x 的后继(后继定义为现有所有数中大于 x 的最小的数)。
516 int get_suc(TreeNode *root, int x) {
517 if(!root)
518 return INF;
519 if(root->val <= x)
520 return get_suc(root->right, x);
521 else
522 return min(root->val, get_suc(root->left, x));
523 }
