118 void rearrangedList(struct ListNode *head) {
119 if(head == NULL || head->next == NULL)
120 return;
121 int len = 0;
122 for(struct ListNode *p = head; p != NULL; p = p->next)
123 len++;
124 int mid = (len + 1) / 2;
125 struct ListNode *a = head;
126 for(int i = 0; i < mid - 1; i++) {
127 a = a->next;
128 }
129 struct ListNode *b = a->next;
130 struct ListNode *c = b->next;
131 a->next = NULL;
132 b->next = NULL;
133 while(c) {
134 struct ListNode *tmp = c->next;
135 c->next = b;
136 b = c;
137 c = tmp;
138 }
139 struct ListNode *p = head;
140 struct ListNode *q = b;

142 auto qq = q->next;
143 // 插入结点
144 q->next = p->next;
