420 int bfs(point start, int **field, int max_x, int max_y) {
421 auto que = deque<point>();
422 que.push_front(start);
423 while(!que.empty()) {
424 const auto p = que.front();
425 que.pop_front();
426 point nexts[] = {
427 point(p.x + 1, p.y, p.step), point(p.x - 1, p.y, p.step),
428 point(p.x, p.y + 1, p.step), point(p.x, p.y - 1, p.step)};
429 for(auto next: nexts) {
430 if(next.x == 0 && next.y == 0) {
431 return next.step;
432 }
433 if(0 <= next.x && next.x <= max_x + 2 && 0 <= next.y && next.y <= max_y + 2 &&
434 field[next.x][next.y] != 2) {
435 if(field[next.x][next.y] == 0) {
436 que.push_front(next);
437 } else {
438 //field[nexts.first][nexts.second]==1
439 next.step++;
440 que.push_back(next);
441 }
442 field[next.x][next.y] = 2;
443 }
444 }
445 }
446 return start.step;

569 int n) {
570 if(level == 0 && !stage) {
571 if(count == 13) {
572 printf("13");

574 return count;
575 }
576
577 pair<int, int> nexts[4] = {pair(x - 1, y), pair(x + 1, y), pair(x, y - 1),
578 pair(x, y + 1)};
579 //复制状态
580 bool picked_cpy[5][5];
581 memcpy(picked_cpy, picked, sizeof picked_cpy);
582
583 int max = 0;
584 picked_cpy[x][y] = true;
585
586 for(const auto next: nexts) {
587 if(0 <= next.first && next.first < n && 0 <= next.second && next.second < n &&
588 !picked_cpy[next.first][next.second]) {
589 int res = 0;
590 if(stage && horseshoes[next.first][next.second] == '(') {
591 res = dfs(true, horseshoes, picked_cpy, count + 1, level + 1, next.first, next.second, n);
592 } else if(stage && horseshoes[next.first][next.second] == ')') {
593 res = dfs(false, horseshoes, picked_cpy, count + 1, level - 1, next.first, next.second, n);
594 } else if(!stage && horseshoes[next.first][next.second] == ')') {
595 res = dfs(false, horseshoes, picked_cpy, count + 1, level - 1, next.first, next.second, n);
596 }
597 if(max < res) {
598 max = res;
599 }
600 }
