290std::vector<int> huffman::generate_bounded_normal_ints(int n) {
291 static std::random_device rd;
292 static std::mt19937 gen(rd());
293
294 std::uniform_int_distribution<int> mu_dist(0, 25);
295 double mean = mu_dist(gen);
296
297 double dist_to_closest_edge = std::min(mean - 0.0, 25.0 - mean);
298 double stddev = std::max(dist_to_closest_edge * 3, 1.0);
299
300 std::normal_distribution<double> norm_dist(mean, stddev);
301 std::vector<int> results;
302 results.reserve(n);
303
304 for(int i = 0; i < n; ++i) {
305 double val;
306 do {
307 val = norm_dist(gen);
308 } while(val < 0 || val > 25);
309
310 results.push_back(static_cast<int>(std::round(val)));
311 }
312 return results;
313}
