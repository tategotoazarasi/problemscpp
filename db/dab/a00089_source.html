228TEST(AVLNodeTest, DeeperInsertionAndFind) {
229 // Create nodes
230 AVLNode<int> root(20);
231 AVLNode<int> node10(10);
232 AVLNode<int> node30(30);
233 AVLNode<int> node5(5);
234 AVLNode<int> node15(15);
235 AVLNode<int> node25(25);
236 AVLNode<int> node35(35);
237
238 // Insert them
239 root.insert(&node10);
240 root.insert(&node30);
241 root.insert(&node5);
242 root.insert(&node15);
243 root.insert(&node25);
244 root.insert(&node35);
245
246 /*
247 * The resulting tree structure should be:
248 * 20 (h=3)
249 * / \
250 * 10(h=2) 30(h=2)
251 * / \ / \
252 * 5 15 25 35 (all h=1)
253 */
254
255 // Verify structure
256 ASSERT_EQ(root.get_left(), &node10);
257 ASSERT_EQ(root.get_right(), &node30);
258 ASSERT_EQ(node10.get_parent(), &root);
259 ASSERT_EQ(node30.get_parent(), &root);
260
261 ASSERT_EQ(node10.get_left(), &node5);
262 ASSERT_EQ(node10.get_right(), &node15);
263 ASSERT_EQ(node5.get_parent(), &node10);
264 ASSERT_EQ(node15.get_parent(), &node10);
265
266 ASSERT_EQ(node30.get_left(), &node25);
267 ASSERT_EQ(node30.get_right(), &node35);
268 ASSERT_EQ(node25.get_parent(), &node30);
269 ASSERT_EQ(node35.get_parent(), &node30);
270
271 // Verify heights
272 ASSERT_EQ(root.get_height(), 3);
273 ASSERT_EQ(node10.get_height(), 2);
274 ASSERT_EQ(node30.get_height(), 2);
275 ASSERT_EQ(node5.get_height(), 1);
276 ASSERT_EQ(node15.get_height(), 1);
277 ASSERT_EQ(node25.get_height(), 1);
278 ASSERT_EQ(node35.get_height(), 1);
279
280 // Verify find method (for existing values)
281 ASSERT_EQ(root.find(20), &root);
