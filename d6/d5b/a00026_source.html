450 namespace the_robotic_rush {
451 int main(istream &cin, ostream &cout) {
452 int t;
453 cin >> t;
454 while(t--) {
455 int n, m, k;
456 cin >> n >> m >> k;
457 vector<int> a(n);
458 vector<int> b(m);
459 vector<node> nodes(n + m);
460 int nodes_i = 0;
461 string instructions;
462 for(int i = 0; i < n; i++) {
463 cin >> a[i];
464 nodes[nodes_i++] = node{true, a[i]};
465 }
466 for(int i = 0; i < m; i++) {
467 cin >> b[i];
468 nodes[nodes_i++] = node{false, b[i]};
469 }
470 ranges::sort(nodes, [](const node &a, const node &b) {
471 return a.pos < b.pos;
472 });
473 int lmax = INT_MIN;
474 int rmin = INT_MAX;
475 for(auto &node: nodes) {
476 if(!node.robot) {
477 lmax = max(lmax, node.pos);
478 }
479 node.lmax = lmax;
480 }
481 for(int i = nodes.size() - 1; i >= 0; i--) {
482 if(!nodes[i].robot) {
483 rmin = min(rmin, nodes[i].pos);
484 }
485 nodes[i].rmin = rmin;
486 }
487 cin >> instructions;
488 vector<int> r_max(k);
489 int rmax = 0;
490 lmax = 0;
491 vector<int> l_max(k);
492 int pos = 0;
493 for(int i = 0; i < instructions.size(); i++) {
494 if(instructions[i] == 'L') {
495 pos--;
496 } else {
497 pos++;
498 }
499 rmax = max(rmax, pos);
500 lmax = max(lmax, -pos);
501 r_max[i] = rmax;
502 l_max[i] = lmax;
503 }
504 vector<int> drop(k, 0);
505 for(auto &node: nodes) {
506 if(node.robot) {
507 int dist_to_right = (node.rmin == INT_MAX) ? INT_MAX : (node.rmin - node.pos);
508 int dist_to_left = (node.lmax == INT_MIN) ? INT_MAX : (node.pos - node.lmax);
509 auto r_exceed = std::lower_bound(r_max.begin(), r_max.end(), dist_to_right);
510 auto l_exceed = std::lower_bound(l_max.begin(), l_max.end(), dist_to_left);
511 int min_d = min(std::distance(r_max.begin(), r_exceed), std::distance(l_max.begin(), l_exceed));
512 if(min_d < drop.size()) {
513 drop[min_d]++;
514 }
515 }
516 }
517 int current = n;
518 for(int i = 0; i < k; i++) {
519 current -= drop[i];
520 cout << current << ' ';
521 }
522 cout << endl;
523 }
524 return 0;
525 }
