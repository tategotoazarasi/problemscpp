341 namespace prefix_max {
342 int main(istream &cin, ostream &cout) {
343 int t;
344 cin >> t;
345 while(t--) {
346 int n;
347 cin >> n;
348 vector<int> vec(n);
349 for(int i = 0; i < n; i++) {
350 cin >> vec[i];
351 }
352 vector<int> pref_max(n);
353 int maximuim = vec[0];
354 for(int i = 0; i < n; i++) {
355 maximuim = max(vec[i], maximuim);
356 pref_max[i] = maximuim;
357 }
358 vector<int> pref_sum(n);
359 int sum = 0;
360 for(int i = 0; i < n; i++) {
361 sum += pref_max[i];
362 pref_sum[i] = sum;
363 }
364 int ans = pref_sum[n - 1];
365 for(int i = 1; i < n; i++) {
366 int j = 0;
367 while(pref_max[j] < vec[i]) {
368 j++;
369 }
370 int val = vec[j] * j + (pref_sum[n - 1] - ((j - 1 >= 0) ? pref_sum[j - 1] : 0));
371 ans = max(ans, val);
372 }
373 cout << ans << endl;
374 }
375 return 0;
376 }
