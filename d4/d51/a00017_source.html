13 vector<int> Solution::match(vector<vector<int>> s2c, vector<vector<int>> c2s) {
14 int N = s2c.size();
15 students = vector<Student *>();
16 companies = vector<Company *>();
17 for(int i = 0; i < N; i++) {
18 Student *stu = new Student(i);
19 stu->c_priority = unordered_map<Company *, int>();
20 stu->priority_c = unordered_map<int, Company *>();
21 students.emplace_back(stu);
22 Company *com = new Company(i);
23 com->priority = unordered_map<Student *, int>();
24 companies.emplace_back(com);
25 }
26 for(int i = 0; i < N; i++) {
27 for(int j = 0; j < N; j++) {
28 students[i]->c_priority[companies[j]] = s2c[i][j];
29 students[i]->priority_c[s2c[i][j]] = companies[j];
30 companies[i]->priority[students[j]] = c2s[i][j];
31 }
32 }
33 bool flag = true;
34 while(flag) {
35 flag = false;
36 for(int i = 0; i < N; i++) {
37 auto stu = students[i];
38 if(stu->matched_company == nullptr) {
39 flag = true;
40 stu->priority_c[stu->current_priority++]->proposed_students.insert(stu);
41 }
42 }
43
44 for(int i = 0; i < N; i++) {
45 auto com = companies[i];
46 Student *current_matched_student = com->matched_student;
47 if(com->proposed_students.size() > 0) {
48 int min_priority = com->matched_student ? com->priority[com->matched_student] : INT_MAX;
49 Student *max_p_student = com->matched_student;
50 for(auto stu: com->proposed_students) {
51 if(com->priority[stu] < min_priority) {
52 min_priority = com->priority[stu];
53 max_p_student = stu;
54 flag = true;
55 }
56 }
57 if(flag) {
58 if(current_matched_student != nullptr) {
59 current_matched_student->matched_company = nullptr;
60 }
61 com->matched_student = max_p_student;
62 max_p_student->matched_company = com;
63 }
64 com->proposed_students.clear();
65 }
66 }
67 }
68 vector<int> ans = vector<int>(N, -1);
69 for(int i = 0; i < N; i++) {
70 ans[i] = students[i]->matched_company->id;
71 }
72 return ans;
73 }
