20 vector<int> Solution::match(vector<vector<int>> s2c, vector<vector<int>> c2s) {
21 int N = s2c.size();
22 students = vector<Student *>();
23 companies = vector<Company *>();
24 for(int i = 0; i < N; i++) {
25 Student *stu = new Student(i);
26 stu->c_priority = unordered_map<Company *, int>();
27 stu->priority_c = unordered_map<int, Company *>();
28 students.emplace_back(stu);
29 Company *com = new Company(i);
30 com->priority = unordered_map<Student *, int>();
31 companies.emplace_back(com);
32 }
33 for(int i = 0; i < N; i++) {
34 for(int j = 0; j < N; j++) {
35 students[i]->c_priority[companies[j]] = s2c[i][j];
36 students[i]->priority_c[s2c[i][j]] = companies[j];
37 companies[i]->priority[students[j]] = c2s[i][j];
38 }
39 }
40 bool flag = true;
41 while(flag) {
42 flag = false;
43 for(int i = 0; i < N; i++) {
44 auto stu = students[i];
45 if(stu->matched_company == nullptr) {
46 flag = true;
47 stu->priority_c[stu->current_priority++]->proposed_students.insert(stu);
48 }
49 }
50
51 for(int i = 0; i < N; i++) {
52 auto com = companies[i];
53 Student *current_matched_student = com->matched_student;
54 if(com->proposed_students.size() > 0) {
55 int min_priority = com->matched_student ? com->priority[com->matched_student] : INT_MAX;
56 Student *max_p_student = com->matched_student;
57 for(auto stu: com->proposed_students) {
58 if(com->priority[stu] < min_priority) {
59 min_priority = com->priority[stu];
60 max_p_student = stu;
61 flag = true;
62 }
63 }
64 if(flag) {
65 if(current_matched_student != nullptr) {
66 current_matched_student->matched_company = nullptr;
67 }
68 com->matched_student = max_p_student;
69 max_p_student->matched_company = com;
70 }
71 com->proposed_students.clear();
72 }
73 }
74 }
75 vector<int> ans = vector<int>(N, -1);
76 for(int i = 0; i < N; i++) {
77 ans[i] = students[i]->matched_company->id;
78 }
79 return ans;
80 }

149 void cardset::insert(card *c) {
150 if(this->mask == (unsigned short) (-1)) {
151 this->mask = calc_mask(c, *this->cards.begin());
152 }
153 this->cards.insert(c);
154 this->cnt++;
155 if(this->cnt == 3) {
156 vector<int> vec = {};
157 for(auto &c: this->cards) {
158 vec.emplace_back(c->id);

160 std::sort(vec.begin(), vec.end());
161 ans.emplace_back(vec);

166 if(s->mask == (unsigned short) (-1)) {
167 return true;
168 }
169 for(auto &sc: s->cards) {
170 if(calc_mask(sc, c) != s->mask) {
171 return false;
172 }
173 }
174 return true;

363 struct pair_hash {
364 size_t operator()(const pair<int, int> &p) const {
365 return hash<int>()(p.first) ^ (hash<int>()(p.second) << 1);
366 }
367 };
368 struct pair_equal {
369 bool operator()(const pair<int, int> &a, const pair<int, int> &b) const {
370 return a.first == b.first && a.second == b.second;
371 }
372 };
373 int main(istream &cin, ostream &cout) {
374 cin >> n >> m;
375 vector<vector<int>> grid(n, vector<int>(m));
376 for(int i = 0; i < n; i++) {
377 string line;
378 cin >> line;
379 istringstream iss(line);
380 for(int j = 0; j < m; j++) {
381 char c;
382 iss >> c;
383 grid[i][j] = c - '0';
384 }
385 }
386 unordered_set<pair<int, int>, pair_hash, pair_equal> visited{};
387 auto comp = [](const state &a, const state &b) {
388 if(a.step != b.step) {
389 return a.step > b.step;
390 }
391 int dist_a = (n - a.x - 1) + (m - a.y - 1);
392 int dist_b = (n - b.x - 1) + (m - b.y - 1);
393 return dist_a > dist_b;
394 };
395 priority_queue<state, vector<state>, decltype(comp)> pq{};
396 pq.push(state{0, 0, 0});
397 while(!pq.empty()) {
398 auto top = pq.top();
399 pq.pop();
400 int x = top.x;
401 int y = top.y;
402 if(visited.count({x, y})) {
403 continue;
404 }
405 visited.insert(make_pair(x, y));
406 if(x == n - 1 && y == m - 1) {
407 cout << top.step;
408 return 0;
409 }
410 int step = grid[x][y];
411 pair<int, int> next[4] = {
412 make_pair(x + step, y),
413 make_pair(x - step, y),
414 make_pair(x, y + step),
415 make_pair(x, y - step)};
416 for(auto [next_x, next_y]: next) {
417 if(next_x == n - 1 && next_y == m - 1) {
418 cout << top.step + 1;
419 return 0;
420 }
421 if(next_x >= 0 && next_x < n && next_y >= 0 && next_y < m && grid[next_x][next_y] != 0) {
422 pq.push(state{next_x, next_y, top.step + 1});
423 }
424 }
425 }
426 cout << -1;
427 return 0;
428 }
