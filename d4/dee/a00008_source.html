181BigInt BigInt::operator+(const BigInt &bi) const {
182 if(this->positive && !bi.positive) {
183 return (*this) - (-bi);
184 }
185 if(!this->positive && bi.positive) {
186 return bi - (-(*this));

190 for(long long i = 0; i < max(this->get_size(), bi.get_size()) || carry != 0; i++) {
191 unsigned short this_num = (static_cast<long long>(this->get_size()) - i - 1) >= 0 ? (*this)[this->get_size() - i - 1] : 0;
192 unsigned short bi_num = (static_cast<long long>(bi.get_size()) - i - 1) >= 0 ? bi[bi.get_size() - i - 1] : 0;
193 unsigned short sum = this_num + bi_num;
194 sum += carry;
195 carry = sum / 10;
