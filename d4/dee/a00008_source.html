168BigInt BigInt::operator+(const BigInt &bi) const {
169 if(this->positive && !bi.positive) {
170 return *this - -bi;
171 }
172 if(!this->positive && bi.positive) {
173 return bi - -*this;
174 }
175 vector<unsigned short> v;
176 unsigned short carry = 0;
177 for(long long i = 0; i < max(this->get_size(), bi.get_size()) || carry != 0; i++) {
178 const unsigned short this_num = i < this->get_size() ? (*this)[i] : 0;
179 const unsigned short bi_num = i < bi.get_size() ? bi[i] : 0;
180 unsigned short sum = this_num + bi_num;
181 sum += carry;
182 carry = sum / 10;
183 sum %= 10;
184 v.push_back(sum);
185 }
186 auto ret = BigInt(v, this->positive);
187 return ret;
188}

190BigInt BigInt::operator-(const BigInt &bi) const {
191 if(this->positive != bi.positive) {
192 return *this + -bi;
193 }
194 if(!this->positive) {
195 return -bi - -*this;
196 }
197 if(*this < bi) {
198 return -(bi - *this);
199 }
200 vector<unsigned short> v;
201 unsigned short borrow = 0;
202 for(long long i = 0; i < max(this->get_size(), bi.get_size()); i++) {
203 short this_num = i < this->get_size() ? (*this)[i] : 0;
204 const short bi_num = i < bi.get_size() ? bi[i] : 0;
205 this_num -= borrow;
206 short diff = this_num - bi_num;
207 borrow = 0;
208 while(diff < 0) {
209 borrow++;
210 diff += 10;

212 v.push_back(diff);
