513 const int curMove = turns % 2 == 0 ? mouse : cat;

515 int result = defaultResult;
516 for(int next: graph[curMove]) {
517 if(curMove == cat && next == 0) {

761 bool Solution::dfs(unsigned long long n1, unsigned long long n2, const char *nums, unsigned short length, unsigned short current) {
762 const auto sum = to_string(n1 + n2);
763 if(sum.length() > length - current) {
764 //前两位和的位数超过剩余的位数
765 return false;
766 }
767 if(!equal(sum, nums, current, length)) {
768 //不包含下一个数字
769 return false;
770 }
771 if(current + sum.length() == length) {
772 //终止条件
773 return true;

838 bool Solution::isEscapePossible(vector<vector<int>> &blocked, vector<int> &source, vector<int> &target) {
839 const int limit = blocked.size() * (blocked.size() - 1) / 2;

852 if(source_status == 0) {
853 return false;
854 }
855 if(source_status == 1) {

862 unsigned int Solution::search(unordered_set<point, point_hash> &block_set, point *source, point *target, unsigned int limit) {
863 auto pq = priority_queue<point>();
864 pq.push(point(source->x, source->y, 0, target));
865 int count = 0;
866 while(!pq.empty()) {
867 if(count > limit || pq.size() > limit) {
868 return 2;//不包围
869 }
870 count++;
871 const auto p = pq.top();
872 pq.pop();
873 point nexts[] = {point(p.x + 1, p.y, p.distance + 1, target), point(p.x - 1, p.y, p.distance + 1, target), point(p.x, p.y + 1, p.distance + 1, target), point(p.x, p.y - 1, p.distance + 1, target)};
874 for(auto next: nexts) {
875 if(0 <= next.x && next.x < 1000000 && 0 <= next.y && next.y < 1000000 && !block_set.contains(next)) {
876 if(next.x == target->x && next.y == target->y) {
877 return 1;//连通
878 }
879 pq.push(next);
880 block_set.insert(next);
881 }
882 }
883 }
884 return 0;//不连通
885 }
