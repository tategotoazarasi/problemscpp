55 int Solution::distributeCandies(vector<int> &candies) {
56 auto m = map<long, int>();
57 auto ans = set<int>();
58 const int n = static_cast<int>(candies.size()) / 2;
59 int sum = 0;
60 for(const auto candy: candies) {
61 m[candy]++;
62 }
63 while(sum < n) {
64 for(auto candy: m) {
65 if(candy.second != 0) {
66 if(!ans.contains(static_cast<int>(candy.first))) {
67 ans.insert(static_cast<int>(candy.first));
68 if(ans.size() == m.size()) {
69 return static_cast<int>(ans.size());
70 }
71 }
72 candy.second--;
73 sum++;
74 }
75 if(sum >= n) {
76 break;
77 }
78 }
79 }
80 return static_cast<int>(ans.size());
81 }

148 vector<vector<int>> res;
149 if(a.empty() || b.empty()) {

152 for(int i = 0, j = 0; i < a.size() && j < b.size();) {

157 ++i;

169 if(l[0] == r[0]) {

187 TreeNode *Solution::vlr(TreeNode *node) {
188 if(node->left != nullptr) {
189 if(node->right != nullptr) {
190 auto *tmp = node->right;
191 node->right = vlr(node->left);
192 auto *current = node->right;
193 while(current->right != nullptr) {
194 current = current->right;
195 }
196 current->right = vlr(tmp);
197 node->left = nullptr;
198 } else {
199 node->right = vlr(node->left);
200 node->left = nullptr;
201 }
202 } else {
203 if(node->right != nullptr) {
204 node->right = vlr(node->right);
205 }

212 string Solution::convert(long long index) {

214 if(index % 702 == 0) {
215 prefix--;
216 index = 702;

247 int Solution::minPathSum(vector<vector<int>> &grid) {
248 const unsigned int m = grid.size();
249 const unsigned int n = grid[0].size();
250 auto *dp = new int *[m + 1];
251 for(int i = 0; i <= m; i++) {
252 dp[i] = new int[n + 1];
253 memset(dp[i], INT_MAX, (n + 1) * sizeof(int));
254 }
255 dp[1][1] = grid[0][0];
256 for(int i = 1; i <= m; i++) {
257 for(int j = 1; j <= n; j++) {
258 if(i == 1 && j == 1) {
259 continue;
260 }
261 unsigned int min = dp[i - 1][j];
262 if(dp[i][j - 1] < min) {
263 min = dp[i][j - 1];
264 }
265 dp[i][j] = grid[i - 1][j - 1] + min;
266 }
267 }
268 const auto ans = dp[m][n];
269 for(int i = 0; i <= m; i++) {
270 delete[] dp[i];
271 }
272 delete[] dp;
273 return ans;
274 }

297 int Solution::minDiffInBST(TreeNode *root) {
298 auto vals = vector<int>();
299 auto que = queue<TreeNode *>();
300 que.push(root);
301 while(!que.empty()) {
302 const auto *node = que.front();
303 que.pop();
304 vals.push_back(node->val);
305 if(node->left != nullptr) {
306 que.push(node->left);
307 }
308 if(node->right != nullptr) {
309 que.push(node->right);
310 }
311 }
312 sort(vals.begin(), vals.end());
313 int minimum = INT_MAX;
314 for(int i = 0; i + 1 < vals.size(); i++) {
315 minimum = min(vals[i + 1] - vals[i], minimum);
316 }
317 return minimum;
318 }
