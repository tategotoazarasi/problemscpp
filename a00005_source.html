153 int** g = new int* [N];
154 for (int i = 0; i < N; i++)
155 {
156 g[i] = new int[N];

158 {
159 g[i][j] = 0;

161 }
162 cin >> n;
163 for (int i = 1; i <= n; i++) {
164 g[1][i] = i - 1;
165 g[i][1] = i - 1;
166 }
167 for (int i = 2; i < n; i++) {
168 for (int j = 2; j < n; j++) {
169 if (i == j) {
170 g[i][j] = 0;//对角线为0
171 }
172 else {
173 if (g[i][j - 1] == 0) {//在0后面一格
174 g[i][j] = g[i - 1][j] + 1;//上面一格的值+1
175 }
176 else {//否则
177 g[i][j] = g[i][j - 1] + 1; //前面一格的值+1
178 }
179 if (g[i][j] == n) {//达到上限
180 g[i][j] = 1;//回归到1
181 }
182 }
183 }
184 }
185 for (int i = 1; i < n; i++) {
186 set<int> s;
187 for (int j = 1; j < n; j++) {
188 s.insert(g[i][j]);
189 }
190 for (int j = 0; j < n; j++) {
191 if (!s.count(j)) {
192 g[i][n] = g[n][i] = j; //补最后一列和最后一行的值
193 }
194 }
195 }
