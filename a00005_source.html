495 int bfs(point start, int** field, int max_x, int max_y)
496 {
497 auto que = deque<point>();
498 que.push_front(start);
499 while (!que.empty())
500 {
501 auto p = que.front();
502 que.pop_front();
503 point nexts[] = {
504 point(p.x + 1, p.y, p.step), point(p.x - 1, p.y, p.step),
505 point(p.x, p.y + 1, p.step), point(p.x, p.y - 1, p.step)
506 };
507 for (auto next : nexts)
508 {
509 if (next.x == 0 && next.y == 0)
510 {
511 return next.step;
512 }
513 if (0 <= next.x && next.x <= max_x + 2 && 0 <= next.y && next.y <= max_y + 2 &&
514 field[next.x][next.y] != 2)
515 {
516 if (field[next.x][next.y] == 0)
517 {
518 que.push_front(next);
519 }
520 else
521 {
522 //field[next.first][next.second]==1
523 next.step++;
524 que.push_back(next);
525 }
526 field[next.x][next.y] = 2;
527 }
528 }
529 }
530 return start.step;
