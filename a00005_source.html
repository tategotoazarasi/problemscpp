34 int number;
35 cin >> number;
36 int copy = number;
37 int len = 0;
38 while (copy != 0) {
39 copy /= 10;
40 len++;
41 }
42 vector<int> arr = vector<int>(len);
43 copy = number;
44 for (int i = len - 1; i >= 0; i--) {
45 arr[i] = copy % 10;
46 copy /= 10;
47 }
48
49 int sum = static_cast<int>(pow(2, len));
50 for (int i = 0; i < len; i++) {
51 if (arr[i] > 1) {
52 break;
53 } else if (arr[i] == 0) {
54 sum -= static_cast<int>(pow(2, len - i - 1));
55 }
56 }
