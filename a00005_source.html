144 int acwing4204::main(istream &cin, ostream &cout) {
145 int n;
146 const int N = 1000;
147 int g[N][N];
148 cin >> n;
149 for (int i = 1; i <= n; i++) {
150 g[1][i] = i - 1;
151 g[i][1] = i - 1;
152 }
153 for (int i = 2; i < n; i++) {
154 for (int j = 2; j < n; j++) {
155 if (i == j) {
156 g[i][j] = 0;//对角线为0
157 } else {
158 if (g[i][j - 1] == 0) {//在0后面一格
159 g[i][j] = g[i - 1][j] + 1;//上面一格的值+1
160 } else {//否则
161 g[i][j] = g[i][j - 1] + 1; //前面一格的值+1
162 }
163 if (g[i][j] == n) {//达到上限
164 g[i][j] = 1;//回归到1
165 }
166 }
167 }
168 }
169 for (int i = 1; i < n; i++) {
170 set<int> s;
171 for (int j = 1; j < n; j++) {
172 s.insert(g[i][j]);
173 }
174 for (int j = 0; j < n; j++) {
175 if (!s.count(j)) {
176 g[i][n] = g[n][i] = j; //补最后一列和最后一行的值
177 }
178 }
179 }
180
181 //输出
182 for (int i = 1; i <= n; i++) {
183 for (int j = 1; j <= n; j++) {
184 cout << g[i][j];
185 if (j != n) {
186 cout << " ";
187 }
188 }
189 if (i != n) {
190 cout << endl;
191 }
192 }
193 return 0;
194 }
