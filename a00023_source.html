14 bool TreeNode::operator==(const TreeNode &node) const {
15 if (this->left != nullptr && node.left == nullptr) {
16 return false;
17 }
18 if (this->left == nullptr && node.left != nullptr) {
19 return false;
20 }
21 if (this->right != nullptr && node.right == nullptr) {
22 return false;
23 }
24 if (this->right == nullptr && node.right != nullptr) {
25 return false;
26 }
27 if (this->left != nullptr && node.left != nullptr && (*this->left) != (*node.left)) {
28 return false;
29 }
30 if (this->right != nullptr && node.right != nullptr && (*this->right) != (*node.right)) {
31 return false;
32 }
33 return this->val == node.val;
34 }

65 void TrieNode::insert(const string &str) {
66 auto node = this->nexts[str[0] - 'a'];
67 if ((node) == nullptr) {
68 node = new TrieNode(str[0]);
69 this->nexts[str[0] - 'a'] = node;
70 }
71 if (str.length() == 1) {
72 node->is_end = true;
73 return;
74 } else return node->insert(str.substr(1));
75 }

77 bool TrieNode::dfs(TrieNode *root, const string &str, int start, bool flag) {
78 if (this->ch == 0) {//根节点
79 auto node = this->nexts[str[start] - 'a'];
80 if (node == nullptr) {
81 return false;
82 }
83 if (node->dfs(root, str, start, flag)) {
84 return true;
85 }
86 return false;
87 } else {//非根节点
88//到一个单词结束处
89 if (this->is_end) {
90 if (start == str.length() - 1) {
91 return flag;
92 }
93 auto res = root->dfs(root, str, start + 1, true);
94 if (res) {
95 return true;
96 }
97 }
98 TrieNode *node = nullptr;
99 if (str[start + 1] - 'a' >= 0) {
100 node = this->nexts[str[start + 1] - 'a'];
101 }
102 return node != nullptr && node->dfs(root, str, start + 1, flag);
103 }

119 string Solution::convertToTitle(int columnNumber) {
120 string ans = string();
121 bool round = false;
122 while (columnNumber != 0) {
123 char ch;
124 if (round) {
125 ch = static_cast<char>(columnNumber % 26 + 63);
126 round = false;
127 } else ch = static_cast<char>(columnNumber % 26 + 64);
128 if (ch == '@' && columnNumber >= 26) {
129 ch = 'Z';
130 round = true;
131 } else if (ch == '?' && columnNumber >= 26) {
132 ch = 'Y';
133 round = true;
134 }
135 if ('A' <= ch && ch <= 'Z') {
136 ans.insert(0, 1, ch);
137 }
138 columnNumber /= 26;
139 }
140 return ans;
141 }

165 int Solution::countQuadruplets(vector<int> &nums) {
166 int sum = 0;
167 for (int i = 3; i < nums.size(); i++) {
168 for (int j = 0; j < i - 2; j++) {
169 if (j == i) {
170 continue;
171 }
172 for (int k = j + 1; k < i - 1; k++) {
173 if (k == i) {
174 continue;
175 }
176 for (int l = k + 1; l < i; l++) {
177 if (l == i) {
178 continue;
179 }
180 if (nums[k] + nums[j] + nums[l] == nums[i]) {
181 sum++;
182 }
183 }
184 }
185 }
186 }
187 return sum;

192 bool Solution::isNStraightHand(vector<int> &hand, int groupSize) {
193 if (hand.size() % groupSize != 0) {
194 return false;
195 }
196 if (groupSize == 1) {
197 return true;
198 }
199 sort(hand.begin(), hand.end());
200 auto len = hand.size() / groupSize;
201 for (int i = 0; i < len; i++) {
202 int current = *hand.begin();
203 hand.erase(hand.begin());
204 for (int j = 1; j < groupSize; j++) {
205 auto next = find(hand.begin(), hand.end(), current + 1);
206 if (next == hand.end()) {
207 return false;
208 } else {
209 current = *next;
210 hand.erase(next);
211 }
212 }
213 }
214 return true;

329 if (bank.size() == 1) {
330 return 0;

332 int count = 0;

336 int count_j = 0;

377 namespace maximum_employees_to_be_invited_to_a_meeting {
378 int Solution::maximumInvitations(vector<int> &favourite) {
379 int n = static_cast<int>(favourite.size());
380 int ans = 0;
381 vector<int> v(n), d(n), dp(n, 1);
382 for (int i = 0; i < n; i += 1) {
383 if (not v[i]) {
384 vector<int> w;
385 int x = i;
386 while (not v[x]) {
387 w.push_back(x);
388 v[x] = 1;
389 x = favourite[x];
390 }
391 for (int j = 0; j < w.size(); j += 1) {
392 if (w[j] == x) {
393 ans = max((int) w.size() - j, ans);
394 }
395 }
396 }
397 }
398 queue<int> q;
399 for (int i = 0; i < n; i += 1) {
400 d[favourite[i]] += 1;
401 }
402 for (int i = 0; i < n; i += 1) {
403 if (d[i] == 0) {
404 q.push(i);
405 }
406 }
407 while (not q.empty()) {
408 int u = q.front();
409 q.pop();
410 dp[favourite[u]] = max(dp[favourite[u]], dp[u] + 1);
411 if (not(d[favourite[u]] -= 1)) {
412 q.push(favourite[u]);
413 }
414 }
415 int sum = 0;
416 for (int i = 0; i < n; i += 1) {
417 if (favourite[favourite[i]] == i and favourite[i] > i) {
418 sum += dp[i] + dp[favourite[i]];
419 }
420 }
421 return max(ans, sum);
422 }
