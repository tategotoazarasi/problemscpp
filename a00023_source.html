13 vector<string> Solution::findAllConcatenatedWordsInADict(vector<string> &words) {
14 sort(words.begin(), words.end(), [&](const string &a, const string &b) {
15 return a.size() < b.size();
16 });
17 auto ans = vector<string>();
18 auto node = TrieNode(0);
19 for (const string &word: words) {
20 if (word.length() == 0) {
21 continue;
22 }
23 if (node.dfs(&node, word, 0, false)) {
24 ans.push_back(word);
25 } else {
26 node.insert(word);
27 }
28 }
29 return ans;
30 }

37 void TrieNode::insert(const string &str) {
38 auto node = this->nexts[str[0] - 'a'];
39 if ((node) == nullptr) {
40 node = new TrieNode(str[0]);
41 this->nexts[str[0] - 'a'] = node;
42 }
43 if (str.length() == 1) {
44 node->is_end = true;
45 return;
46 } else return node->insert(str.substr(1));
47 }

49 bool TrieNode::dfs(TrieNode *root, const string &str, int start, bool flag) {
50 if (this->ch == 0) {//根节点
51 auto node = this->nexts[str[start] - 'a'];
52 if (node == nullptr) {
53 return false;
54 }
55 if (node->dfs(root, str, start, flag)) {
56 return true;
57 }
58 return false;
59 } else {//非根节点
60 //到一个单词结束处
61 if (this->is_end) {
62 if (start == str.length() - 1) {
63 return flag;
64 }
65 auto res = root->dfs(root, str, start + 1, true);
66 if (res) {
67 return true;
68 }
69 }
70 TrieNode *node = nullptr;
71 if (str[start + 1] - 'a' >= 0) {
72 node = this->nexts[str[start + 1] - 'a'];
73 }
74 return node != nullptr && node->dfs(root, str, start + 1, flag);
75 }
76 }

80 int Solution::titleToNumber(const std::string &columnTitle) {
81 int sum = 0;
82 int length = static_cast<int>(columnTitle.length());
83 for (char c: columnTitle) {
84 sum += static_cast<int>(static_cast<double>(c - 'A' + 1) * pow(26, length-- - 1));
85 }
86 return sum;
87 }

91 string Solution::convertToTitle(int columnNumber) {
92 string ans = string();
93 bool round = false;
94 while (columnNumber != 0) {
95 char ch;
96 if (round) {
97 ch = static_cast<char>(columnNumber % 26 + 63);
98 round = false;
99 } else ch = static_cast<char>(columnNumber % 26 + 64);
100 if (ch == '@' && columnNumber >= 26) {
101 ch = 'Z';
102 round = true;
103 } else if (ch == '?' && columnNumber >= 26) {
104 ch = 'Y';
105 round = true;
106 }
107 if ('A' <= ch && ch <= 'Z') {
108 ans.insert(0, 1, ch);
109 }
110 columnNumber /= 26;
111 }
112 return ans;
113 }

137 int Solution::countQuadruplets(vector<int> &nums) {
138 int sum = 0;
139 for (int i = 3; i < nums.size(); i++) {
140 for (int j = 0; j < i - 2; j++) {
141 if (j == i) {
142 continue;
143 }
144 for (int k = j + 1; k < i - 1; k++) {
145 if (k == i) {
146 continue;
147 }
148 for (int l = k + 1; l < i; l++) {
149 if (l == i) {
150 continue;
151 }
152 if (nums[k] + nums[j] + nums[l] == nums[i]) {
153 sum++;
154 }
155 }
156 }
157 }
158 }
159 return sum;
160 }
