10 vector<string> Solution::findAllConcatenatedWordsInADict(vector<string> &words) {
11 sort(words.begin(), words.end(), [&](const string &a, const string &b) {
12 return a.size() < b.size();
13 });
14 auto ans = vector<string>();
15 TrieNode node = TrieNode(0);
16 for (const string &word: words) {
17 if (word.length() == 0) {
18 continue;
19 }
20 if (node.dfs(&node, word, 0, false)) {
21 ans.push_back(word);
22 } else {
23 node.insert(word);
24 }
25 }
26 return ans;
27 }

34 void TrieNode::insert(const string &str) {
35 auto node = this->nexts[str[0] - 'a'];
36 if ((node) == nullptr) {
37 node = new TrieNode(str[0]);
38 this->nexts[str[0] - 'a'] = node;
39 }
40 if (str.length() == 1) {
41 node->is_end = true;
42 return;
43 } else return node->insert(str.substr(1));
44 }

46 bool TrieNode::dfs(TrieNode *root, const string &str, int start, bool flag) {
47 if (this->ch == 0) {//根节点
48 auto node = this->nexts[str[start] - 'a'];
49 if (node == nullptr) {
50 return false;
51 }
52 if (node->dfs(root, str, start, flag)) {
53 return true;
54 }
55 return false;
56 } else {//非根节点
57 //到一个单词结束处
58 if (this->is_end) {
59 if (start == str.length() - 1) {
60 return flag;
61 }
62 auto res = root->dfs(root, str, start + 1, true);
63 if (res) {
64 return true;
65 }
66 }
67 TrieNode *node = nullptr;
68 if (str[start + 1] - 'a' >= 0) {
69 node = this->nexts[str[start + 1] - 'a'];
70 }
71 return node != nullptr && node->dfs(root, str, start + 1, flag);
72 }
73 }
