14 bool TreeNode::operator==(const TreeNode& node) const {
15 if (this->left != nullptr && node.left == nullptr) {
16 return false;
17 }
18 if (this->left == nullptr && node.left != nullptr) {
19 return false;
20 }
21 if (this->right != nullptr && node.right == nullptr) {
22 return false;
23 }
24 if (this->right == nullptr && node.right != nullptr) {
25 return false;
26 }
27 if (this->left != nullptr && node.left != nullptr && (*this->left) != (*node.left)) {
28 return false;
29 }
30 if (this->right != nullptr && node.right != nullptr && (*this->right) != (*node.right)) {
31 return false;
32 }
33 return this->val == node.val;
34 }

66 void TrieNode::insert(const string& str) {
67 auto node = this->nexts[str[0] - 'a'];
68 if ((node) == nullptr) {
69 node = new TrieNode(str[0]);
70 this->nexts[str[0] - 'a'] = node;
71 }
72 if (str.length() == 1) {
73 node->is_end = true;
74 return;
75 }
76 else return node->insert(str.substr(1));
77 }

79 bool TrieNode::dfs(TrieNode* root, const string& str, int start, bool flag) {
80 if (this->ch == 0) {//根节点
81 auto node = this->nexts[str[start] - 'a'];
82 if (node == nullptr) {
83 return false;
84 }
85 if (node->dfs(root, str, start, flag)) {
86 return true;
87 }
88 return false;
89 }
90 else {//非根节点
91//到一个单词结束处
92 if (this->is_end) {
93 if (start == str.length() - 1) {
94 return flag;
95 }
96 auto res = root->dfs(root, str, start + 1, true);
97 if (res) {
98 return true;
99 }
100 }
101 TrieNode* node = nullptr;
102 if (str[start + 1] - 'a' >= 0) {
103 node = this->nexts[str[start + 1] - 'a'];
104 }
105 return node != nullptr && node->dfs(root, str, start + 1, flag);
106 }

122 string Solution::convertToTitle(int columnNumber) {
123 string ans = string();
124 bool round = false;
125 while (columnNumber != 0) {
126 char ch;
127 if (round) {
128 ch = static_cast<char>(columnNumber % 26 + 63);
129 round = false;
130 }
131 else ch = static_cast<char>(columnNumber % 26 + 64);
132 if (ch == '@' && columnNumber >= 26) {
133 ch = 'Z';
134 round = true;
135 }
136 else if (ch == '?' && columnNumber >= 26) {
137 ch = 'Y';
138 round = true;
139 }
140 if ('A' <= ch && ch <= 'Z') {
141 ans.insert(0, 1, ch);
142 }
143 columnNumber /= 26;
144 }
145 return ans;

171 int Solution::countQuadruplets(vector<int>& nums) {
172 int sum = 0;
173 for (int i = 3; i < nums.size(); i++) {
174 for (int j = 0; j < i - 2; j++) {
175 if (j == i) {
176 continue;
177 }
178 for (int k = j + 1; k < i - 1; k++) {
179 if (k == i) {
180 continue;
181 }
182 for (int l = k + 1; l < i; l++) {
183 if (l == i) {
184 continue;
185 }
186 if (nums[k] + nums[j] + nums[l] == nums[i]) {
187 sum++;
188 }
189 }
190 }
191 }
192 }
193 return sum;

198 bool Solution::isNStraightHand(vector<int>& hand, int groupSize) {
199 if (hand.size() % groupSize != 0) {
200 return false;
201 }
202 if (groupSize == 1) {
203 return true;
204 }
205 sort(hand.begin(), hand.end());
206 auto len = hand.size() / groupSize;
207 for (int i = 0; i < len; i++) {
208 int current = *hand.begin();
209 hand.erase(hand.begin());
210 for (int j = 1; j < groupSize; j++) {
211 auto next = find(hand.begin(), hand.end(), current + 1);
212 if (next == hand.end()) {
213 return false;
214 }
215 else {
216 current = *next;
217 hand.erase(next);
218 }
219 }
220 }
221 return true;
222 }

227 int sum = 0;
228 int max = num;
229 for (int i = 1; i < max; i++) {
230 if (num % i == 0) {
231 sum += i;
232 if (i != 1) {
233 sum += num / i;
234 }
235 max = num / i;
236 }
237 }

276 void Solution::convert(FriendTreeNode* sum_node) {
277 if (sum_node->right != nullptr) {
278 sum_node->right->val = sum_node->val - sum_node->friend_node->val -
279 (sum_node->right->left == nullptr ? 0 : sum_node->right->left->sum);
280 Solution::convert(sum_node->right);

282 if (sum_node->left != nullptr) {
283 sum_node->left->val = sum_node->val + sum_node->left->friend_node->val +

285 Solution::convert(sum_node->left);

311 int Solution::lastRemaining(int n) {
312 int num_amount = n;
313 int loop_cnt = 0;
314 int a0 = 1; //初项
315 int d = 1; //差
316 while (num_amount != 1) {//剩下的数目不为1时
317 if (num_amount % 2 == 1) {// 奇数个数字
318 a0 = a0 + d;
319 }
320 else if (num_amount % 2 == 0) {// 偶数个数字
321 bool left_to_right = (loop_cnt % 2 == 0);
322 if (left_to_right) {
323 a0 = a0 + d;
324 }
325 else
326 a0 = a0;
327 }
328 loop_cnt++;
329 d *= 2;
330 num_amount /= 2;
331 }
332 return a0;
