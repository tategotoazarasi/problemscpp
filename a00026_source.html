52 int Solution::distributeCandies(vector<int> &candies) {
53 auto m = map<long, int>();
54 auto ans = set<int>();
55 const int n = static_cast<int>(candies.size()) / 2;
56 int sum = 0;
57 for(const auto candy: candies) {
58 m[candy]++;
59 }
60 while(sum < n) {
61 for(auto candy: m) {
62 if(candy.second != 0) {
63 if(!ans.contains(static_cast<int>(candy.first))) {
64 ans.insert(static_cast<int>(candy.first));
65 if(ans.size() == m.size()) {
66 return static_cast<int>(ans.size());
67 }
68 }
69 candy.second--;
70 sum++;
71 }
72 if(sum >= n) {
73 break;
74 }
75 }
76 }
77 return static_cast<int>(ans.size());
78 }

156 vector<vector<int>> Solution::Intersection(vector<vector<int>> &a, vector<vector<int>> &b) {
157 vector<vector<int>> res;
158 if(a.empty() || b.empty()) {
159 return res;
160 }
161 for(int i = 0, j = 0; i < a.size() && j < b.size();) {
162 if(is_intersected(a[i], b[j])) {
163 res.emplace_back(vector({i, j}));
164 }
165 if(a[i][1] == b[j][1]) {
166 ++i;
167 ++j;
168 } else if(a[i][1] > b[j][1]) {
169 ++j;
170 } else {
171 ++i;
172 }
173 }
174 return res;

178 if(l[0] == r[0]) {

196 TreeNode *Solution::vlr(TreeNode *node) {
197 if(node->left != nullptr) {
198 if(node->right != nullptr) {
199 auto *tmp = node->right;
200 node->right = vlr(node->left);
201 auto *current = node->right;
202 while(current->right != nullptr) {
203 current = current->right;
204 }
205 current->right = vlr(tmp);
206 node->left = nullptr;
207 } else {
208 node->right = vlr(node->left);
209 node->left = nullptr;
210 }
211 } else {
212 if(node->right != nullptr) {
213 node->right = vlr(node->right);
214 }
215 }
