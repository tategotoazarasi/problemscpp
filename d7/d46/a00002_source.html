56 int Solution::distributeCandies(vector<int> &candies) {
57 auto m = map<long, int>();
58 auto ans = set<int>();
59 const int n = static_cast<int>(candies.size()) / 2;
60 int sum = 0;
61 for(const auto candy: candies) {
62 m[candy]++;
63 }
64 while(sum < n) {
65 for(auto candy: m) {
66 if(candy.second != 0) {
67 if(!ans.contains(static_cast<int>(candy.first))) {
68 ans.insert(static_cast<int>(candy.first));
69 if(ans.size() == m.size()) {
70 return static_cast<int>(ans.size());
71 }
72 }
73 candy.second--;
74 sum++;
75 }
76 if(sum >= n) {
77 break;
78 }
79 }
80 }
81 return static_cast<int>(ans.size());
82 }

161 vector<vector<int>> res;

200 TreeNode *Solution::vlr(TreeNode *node) {
201 if(node->left != nullptr) {
202 if(node->right != nullptr) {
203 auto *tmp = node->right;
204 node->right = vlr(node->left);
205 auto *current = node->right;

207 current = current->right;

209 current->right = vlr(tmp);
210 node->left = nullptr;

212 node->right = vlr(node->left);
213 node->left = nullptr;

260 int Solution::minPathSum(vector<vector<int>> &grid) {
261 const unsigned int m = grid.size();
262 const unsigned int n = grid[0].size();
263 auto *dp = new int *[m + 1];
264 for(int i = 0; i <= m; i++) {
265 dp[i] = new int[n + 1];
266 memset(dp[i], INT_MAX, (n + 1) * sizeof(int));
267 }
268 dp[1][1] = grid[0][0];
269 for(int i = 1; i <= m; i++) {
270 for(int j = 1; j <= n; j++) {
271 if(i == 1 && j == 1) {
272 continue;
273 }
274 unsigned int min = dp[i - 1][j];
275 if(dp[i][j - 1] < min) {
276 min = dp[i][j - 1];
277 }
278 dp[i][j] = grid[i - 1][j - 1] + min;
279 }
280 }
281 return dp[m][n];
282 }

305 int Solution::minDiffInBST(TreeNode *root) {
306 auto vals = vector<int>();
307 auto que = queue<TreeNode *>();
308 que.push(root);
309 while(!que.empty()) {
310 const auto *node = que.front();
311 que.pop();
312 vals.push_back(node->val);
313 if(node->left != nullptr) {
314 que.push(node->left);
315 }
316 if(node->right != nullptr) {
317 que.push(node->right);
318 }
319 }
320 sort(vals.begin(), vals.end());
321 int minimum = INT_MAX;
322 for(int i = 0; i + 1 < vals.size(); i++) {
323 minimum = min(vals[i + 1] - vals[i], minimum);
324 }
325 return minimum;
326 }
